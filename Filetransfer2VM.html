# ========================================================= # PRACTICAL 10 — FILE TRANSFER BETWEEN TWO VIRTUAL MACHINES # STEP 1 — CREATE TWO VIRTUAL MACHINES # ========================================================= # Open VirtualBox # Create: # VM1 → UBUNTU22 # VM2 → UBUNTU2 # Install Ubuntu in both VMs. # ========================================================= # STEP 2 — START BOTH VMs # ========================================================= # Open both virtual machines. # ========================================================= # STEP 3 — CONNECT BOTH VMs TO SAME NETWORK # ========================================================= # VirtualBox # → Settings # → Network # Select: # Attached To → Bridged Adapter # OR # Attached To → NAT Network # Use SAME network mode in both VMs. # ========================================================= # STEP 4 — CHECK IP ADDRESS # RUN IN BOTH VMs # ========================================================= ip a # --------------------------------------------------------- # FIND IP ADDRESS # # Example: # # VM1: # 192.168.1.10 # # VM2: # 192.168.1.20 # --------------------------------------------------------- # ========================================================= # STEP 5 — INSTALL SSH SERVER # RUN IN BOTH VMs # ========================================================= sudo apt update sudo apt install openssh-server # ========================================================= # STEP 6 — START SSH SERVICE # ========================================================= sudo systemctl start ssh # ========================================================= # STEP 7 — ENABLE SSH # ========================================================= sudo systemctl enable ssh # ========================================================= # STEP 8 — CHECK SSH STATUS # ========================================================= sudo systemctl status ssh # --------------------------------------------------------- # OUTPUT SHOULD SHOW: # # active (running) # # PRESS: # q # TO EXIT STATUS SCREEN # --------------------------------------------------------- # ========================================================= # STEP 9 — CREATE SAMPLE FILE IN VM1 # ========================================================= touch test.txt # ========================================================= # STEP 10 — TRANSFER FILE USING SCP # RUN IN VM1 # ========================================================= scp test.txt username@IP_ADDRESS:/home/username/ # --------------------------------------------------------- # EXAMPLE # --------------------------------------------------------- scp test.txt atharva@192.168.1.20:/home/atharva/ # --------------------------------------------------------- # EXPLANATION # --------------------------------------------------------- # scp # Secure copy command. # test.txt # File to transfer. # atharva # Username of VM2. # 192.168.1.20 # IP address of VM2. # /home/atharva/ # Destination folder. # ========================================================= # STEP 11 — ENTER PASSWORD # ========================================================= # Enter VM2 password. # ========================================================= # STEP 12 — VERIFY FILE IN VM2 # ========================================================= ls # --------------------------------------------------------- # OUTPUT # --------------------------------------------------------- # test.txt should appear. # ========================================================= # OPTIONAL — TRANSFER FOLDER # ========================================================= scp -r foldername username@IP_ADDRESS:/home/username/
