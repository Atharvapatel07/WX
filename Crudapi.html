# ========================================================= # PRACTICAL 7 — CRUD API USING NODE.JS + EXPRESS + MONGODB # ========================================================= # --------------------------------------------------------- # STEP 1 — UPDATE UBUNTU # --------------------------------------------------------- sudo apt update # --------------------------------------------------------- # STEP 2 — INSTALL NODEJS & NPM # --------------------------------------------------------- sudo apt install nodejs npm # --------------------------------------------------------- # STEP 3 — CHECK NODE VERSION # --------------------------------------------------------- node -v # --------------------------------------------------------- # STEP 4 — CHECK NPM VERSION # --------------------------------------------------------- npm -v # --------------------------------------------------------- # STEP 5 — CREATE PROJECT FOLDER # --------------------------------------------------------- mkdir crudapp # --------------------------------------------------------- # STEP 6 — OPEN PROJECT FOLDER # --------------------------------------------------------- cd crudapp # --------------------------------------------------------- # STEP 7 — INITIALIZE NODE PROJECT # CREATES package.json FILE # --------------------------------------------------------- npm init -y # --------------------------------------------------------- # STEP 8 — INSTALL EXPRESS PACKAGE # --------------------------------------------------------- npm install express # --------------------------------------------------------- # STEP 9 — INSTALL MONGOOSE PACKAGE # USED TO CONNECT NODEJS WITH MONGODB # --------------------------------------------------------- npm install mongoose # --------------------------------------------------------- # STEP 10 — INSTALL BODY PARSER # USED TO HANDLE JSON DATA # --------------------------------------------------------- npm install body-parser # --------------------------------------------------------- # STEP 11 — CREATE APPLICATION FILE # --------------------------------------------------------- touch app.js # --------------------------------------------------------- # STEP 12 — OPEN VS CODE # --------------------------------------------------------- code . # --------------------------------------------------------- # STEP 13 — FIX CDROM REPOSITORY ERROR # ONLY IF apt update SHOWS cdrom ERROR # --------------------------------------------------------- sudo nano /etc/apt/sources.list # --------------------------------------------------------- # INSIDE FILE: # FIND LINE CONTAINING: # deb cdrom: # OR # file:/cdrom # # ADD # BEFORE THAT LINE # # EXAMPLE: # # deb cdrom: # # SAVE: # Ctrl + O # ENTER # # EXIT: # Ctrl + X # --------------------------------------------------------- # --------------------------------------------------------- # STEP 14 — UPDATE UBUNTU AGAIN # --------------------------------------------------------- sudo apt update # --------------------------------------------------------- # STEP 15 — INSTALL MONGODB # --------------------------------------------------------- sudo apt install mongodb # --------------------------------------------------------- # STEP 16 — START MONGODB SERVICE # --------------------------------------------------------- sudo systemctl start mongodb # --------------------------------------------------------- # STEP 17 — ENABLE MONGODB # --------------------------------------------------------- sudo systemctl enable mongodb # --------------------------------------------------------- # STEP 18 — CHECK MONGODB STATUS # --------------------------------------------------------- sudo systemctl status mongodb # --------------------------------------------------------- # PRESS q TO EXIT STATUS SCREEN # --------------------------------------------------------- # --------------------------------------------------------- # STEP 19 — RUN NODE APPLICATION # --------------------------------------------------------- node app.js # --------------------------------------------------------- # OUTPUT SHOULD SHOW: # # MongoDB Connected # Server Running on Port 3000 # --------------------------------------------------------- # ========================================================= # CRUD API URLs # ========================================================= # --------------------------------------------------------- # CREATE API # METHOD: POST # URL: # --------------------------------------------------------- http://localhost:3000/add # --------------------------------------------------------- # READ API # METHOD: GET # URL: # --------------------------------------------------------- http://localhost:3000/students # --------------------------------------------------------- # UPDATE API # METHOD: PUT # URL: # --------------------------------------------------------- http://localhost:3000/update/ID # --------------------------------------------------------- # DELETE API # METHOD: DELETE # URL: # --------------------------------------------------------- http://localhost:3000/delete/ID # ========================================================= # IMPORTANT VIVA POINTS # ========================================================= # CRUD: # C → CREATE # R → READ # U → UPDATE # D → DELETE # POST → CREATE DATA # GET → FETCH DATA # PUT → UPDATE DATA # DELETE → REMOVE DATA # EXPRESS → NODEJS FRAMEWORK # MONGODB → NOSQL DATABASE # MONGOOSE → CONNECTS NODEJS WITH MONGODB #
